304 void endGame() {
305 Rock->Enabled = false;
306 Paper->Enabled = false;
307 Scissor->Enabled = false;
308 start->Enabled = true;
309
310
311 if (scoreP == scoreD) {
312 lblRound->Text = " ";
313 lblResult->Text = "Tie";
314
315 User::credit += Convert::ToInt32(comboBox->SelectedItem);
316
317 labelBalance->Text = "Credit: " + User::credit.ToString();
318 }
319 else if (scoreP > scoreD) {
320 lblRound->Text = " ";
321 lblResult->Text = "You Win!";
322
323 User::credit += Convert::ToInt32(comboBox->SelectedItem) * 2;
324
325 labelBalance->Text = "Credit: " + User::credit.ToString();
326 }
327 else if (scoreP < scoreD) {
328 lblRound->Text = " ";
329 lblResult->Text = "You Lose!";
330 labelBalance->Text = "Credit: " + User::credit.ToString();
331 }
